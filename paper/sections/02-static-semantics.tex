%% unit
\AXC{}
\RightLabel{(Unit)}
\UIC{$\Delta \ | \ \Gamma \vdash \langle \rangle : \mathbf{unit}$}
\bottomAlignProof
\DisplayProof
\hskip 0.5em
%% var
\AXC{$s \ge 1$}
\RightLabel{(Var)}
\UIC{$\Delta \ | \ \Gamma, x:_s \tau, \Theta \vdash x : \tau$}
\bottomAlignProof
\DisplayProof
\hskip 0.5em
%% fun
\AXC{$\Delta \ | \ \Gamma, x:_1 \tau_0 \vdash e : \tau$}
\RightLabel{($\multimap$ I)}
\UnaryInfC{$\Delta \ | \ \Gamma \vdash \lambda x. e : \tau_0 \multimap \tau $}
\bottomAlignProof
\DisplayProof
\vskip 1em

%% app
\AXC{$\Delta \ | \ \Gamma \vdash e : \tau_0 \multimap \tau$}
\AXC{$\Delta \ | \ \Theta \vdash f : \tau_0 $}
\RightLabel{($\multimap$ E)}
\BinaryInfC{$\Delta \ | \ \Gamma + \Theta \vdash ef : \tau $}
\bottomAlignProof
\DisplayProof
\hskip 0.5em
%% dep prod intro
\AXC{$\Delta \ | \ \Gamma \vdash e : \tau_0$}
\AXC{$\Delta \ | \ \Gamma \vdash f : \tau_1$}
\RightLabel{($\times$ I)}
\BinaryInfC{$\Delta \ | \ \Gamma \vdash \langle e, f \rangle: \tau_0 \times \tau_1$}
\bottomAlignProof
\DisplayProof
\vskip 1em
%%


%% ROW2
%% dep prod elim
\AXC{$\Delta \ | \ \Gamma \vdash e : \tau_1 \times \tau_2$}
\RightLabel{($\times$ E)}
\UIC{$\Delta \ | \ \Gamma \vdash {\pi}_i \ e : \tau_i$}
\bottomAlignProof
\DisplayProof
\hskip 0.5em
%% ind prod intro
\AXC{$\Delta \ | \ \Gamma \vdash e : \tau_0 $}
\AXC{$\Delta \ | \ \Theta \vdash f : \tau_1$}
\RightLabel{($\tensor$ I)}
\BIC{$\Delta \ | \ \Gamma + \Theta \vdash (e, f) : \tau_0 \tensor \tau_1$}
\bottomAlignProof
\DisplayProof
\vskip 1em

%% ind prod elim
\AXC{$\Delta \ | \ \Gamma \vdash e : \tau_0 \tensor \tau_1$ }
\AXC{$\Delta \ | \ \Theta,x:_s \tau_0,y:_s\tau_1 \vdash f: \tau $}
\RightLabel{($\tensor$ E)}
\BIC{$\Delta \ | \ s * \Gamma + \Theta \vdash \letpair (x,y) \ = \ e \ \tin \ f : \tau $}
\bottomAlignProof
\DisplayProof
\hskip 0.5em
%% ind sum intro
\AXC{$\Delta \ | \ \Gamma \vdash e : \tau_0$ }
\RightLabel{($+$ $\text{I}_i$)}
\UIC{$\Delta \ | \ \Gamma \vdash \mathbf{in}_i \ e : \tau_0 + \tau_1$}
\bottomAlignProof
\DisplayProof
\vskip 1em

% sum elim
\AXC{$\Delta \ | \ \Gamma \vdash e : \tau_0+\tau_1$}
\AXC{$\Delta \ | \ \Theta, x:_s \tau_0 \vdash f_1 : \tau$ \qquad
$\Delta \ | \ \Theta, x:_s \tau_1 \vdash f_2: \tau$}
\RightLabel{($+$ E)}
\AXC{$s > 0$}
\TIC{$\Delta + \Delta \ | \ s * \Gamma + \Theta \vdash \mathbf{case} \ e \ \mathbf{of} \ (\mathbf{in}_1 x.f_1 \ | \ \mathbf{in}_2 x.f_2) : \tau$}
\bottomAlignProof
\DisplayProof
\vskip 1em

% box elim
\AXC{$\Delta \ | \ \Gamma \vdash e : {!_s \tau_0}$}
\AXC{$\Delta \ | \ \Theta, x:_{t*s} \tau_0 \vdash f : \tau$}
\RightLabel{($!$ E)}
\BIC{$\Delta \ | \ t * \Gamma + \Theta \vdash \letcobind x = e \ \tin \ f : \tau$}
\bottomAlignProof
\DisplayProof
\hskip 0.5em
% ops
% \AXC{$\Delta \ | \ \Gamma \vdash e : \tau_0$}
\AXC{$\{ \mathbf{op} :\tau \} \in \Sigma$}
\RightLabel{(Op)}
\UIC{$\Delta \ | \ \Gamma \vdash \mathbf{op} : \tau$}
\bottomAlignProof
\DisplayProof
\vskip 1em

%% ROW 5

%%% ROW 6

% let 
\AXC{$\Delta \ | \ \Gamma \vdash e :  \tau_0$}
\AXC{$\Delta \ | \ \Theta, x:_{s} \tau \vdash f : \tau$}
\RightLabel{(Let)}
\BIC{$\Delta \ | \ s * \Gamma + \Theta \vdash \letassign x = e \ \tin \ f : \tau$}
\bottomAlignProof
\DisplayProof
\hskip 0.5em
%% const
\AXC{$k, k_0, k_1 \in \textit{num}$}
\AXC{$k_0 \leq k \leq k_1$}
\RightLabel{(Const)}
\BIC{$\Delta \ | \ \Gamma \vdash k : \num_{(k_0, k_1)}$}
\bottomAlignProof
\DisplayProof
\vskip 1em

%% widen
\AXC{$(k_0, k_1) \sqsubseteq (k'_0, k'_1)$}
\AXC{$\Delta \ | \ \Gamma \vdash e : \num_{(k_0, k_1)}$}
\RightLabel{(Widen)}
\BIC{$\Delta \ | \ \Gamma \vdash e : \num_{(k'_0, k'_1)}$}
\bottomAlignProof
\DisplayProof
\hskip 0.5em
% box intro
\AXC{$\Delta \ | \ \Gamma \vdash e : \tau$ }
\RightLabel{($!$ I)}
\UIC{$\Delta \ | \ s * \Gamma \vdash [e] : {!_s \tau}$}
\bottomAlignProof
\DisplayProof
\vskip 1em

%%% ROW 7

%% subsumption
\AXC{$\Delta \ | \ \Gamma \vdash e :  M_q \tau$}
\AXC{$r \ge q$}
\RightLabel{(Subsumption)}
\BIC{$\Delta \ | \ \Gamma \vdash e :  M_{r} \tau$}
\bottomAlignProof
\DisplayProof
\hskip 0.5em
% factor
\AXC{$\Delta \ | \ \Gamma \vdash e : (M_q \tau_0) \times (M_r \tau_1)$}
\RightLabel{(Factor)}
\UIC{$\Delta \ | \ \Gamma \vdash \factor \ e : M_{max(q,r)} (\tau_0 \times \tau_1)$}
\bottomAlignProof
\DisplayProof
\vskip 1em
%% return
\AXC{$\Delta \ | \ \Gamma \vdash e : \tau$}
\RightLabel{(Ret)}
\UIC{$\Delta \ | \ \Gamma \vdash \ret e : M_0 \tau$}
\bottomAlignProof
\DisplayProof
\hskip 0.5em
%% RND
\AXC{$\Delta \ | \ \Gamma \vdash e : \num$}
\RightLabel{(Rnd)}
\UIC{$\Delta \ | \ \Gamma \vdash \rnd \ e : M_u \ \num$}
\bottomAlignProof
\DisplayProof
\vskip 1em


%%% ROW 8


% let-bind
\AXC{$\Delta \ | \ \Gamma \vdash e : M_r \tau_0$}
\AXC{$\Delta \ | \ \Theta, x:_{s} \tau_0 \vdash f : M_{q} \tau$}
\RightLabel{($M_u$ E)}
\BIC{$\Delta \ | \ s * \Gamma + \Theta \vdash \letbind x = e \ \tin \ f : M_{s*r+q} \tau$}
\bottomAlignProof
\DisplayProof

\vskip 1em
% bnd-var
\AXC{}
\RightLabel{($\textbf{bnd-var}$)}
\UIC{$\Delta, \epsilon : \textbf{bnd} \vdash \epsilon : \textbf{bnd}$}
\bottomAlignProof
\DisplayProof


\vskip 1em

\AXC{$\Delta, \epsilon : \textbf{bnd} \ | \ \Gamma \vdash e : \tau$}
\AXC{$\epsilon \not\in FTV(\Gamma)$}
\RightLabel{($\forall$-I)}
\BIC{$\Delta \ | \ \Gamma \vdash \Lambda \epsilon . e : \forall \epsilon . \tau$}
\bottomAlignProof
\DisplayProof
\hskip 0.5em
\AXC{$\Delta \ | \ \Gamma \vdash e : \forall \epsilon . \tau$}
\AXC{$\Delta \vdash i : \textbf{bnd}$}
\RightLabel{($\forall$-E)}
\BIC{$\Delta \ | \ \Gamma \vdash e~\{i\} : \tau[i/\epsilon]$}
\bottomAlignProof
\DisplayProof

\vskip 1em

\AXC{$\Delta \vdash b_0 : \textbf{bnd}$}
\AXC{$\ldots$}
\AXC{$\Delta \vdash b_c : \textbf{bnd}$}
\AXC{$\textbf{iop}_c \in \Sigma_{\textbf{bnd}}$}
\RightLabel{($\textbf{bnd}$-\textbf{op})}
\QIC{$\Delta \vdash \textbf{iop}_c(i_0, \ldots, i_c) : \textbf{bnd}$}
\bottomAlignProof
\DisplayProof

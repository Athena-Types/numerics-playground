  \letdestruct (x, y) = (v, w)\ \tin e &\mapsto e[v/x][w/y] \\

	\letbind x = \ret v \ \tin \ e &\mapsto e[v/x] 
\end{aligned}
\end{equation*}

	\qquad \mathbf{case} \ (\mathbf{in}_i \ v) \ \mathbf{of} \ (\mathbf{in}_1 \ x.e_1 \ | \ \mathbf{in}_2 \ x.e_2 )  \mapsto e_i[v/x]
